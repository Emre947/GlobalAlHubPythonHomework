kullanıcı=[]
for i in range(1):
    ad=str(input('ad: '))
    soyad=str(input('soyad: '))
    yaş=int(input('yaş: '))
    if yaş<18:
        print('dışarı çıkamazsın çünkü tehlikeli.')
    elif yaş>18:
        print('dışarı çıkabilirsin.')
    doğum_yılı=int(input('doğum yılı: '))
    kullanıcı.append([ad,soyad,yaş,doğum_yılı])
print('kullanıcı={},{},{},{}'.format(kullanıcı[0][0][0:5],kullanıcı[0][1][0:5],kullanıcı[0][2],kullanıcı[0][3]))
